// =============================================================
// ⚙️ 사용자 설정 (USER SETTINGS) — 이 부분만 수정하면 됩니다
// =============================================================

// 📧 보고서를 받을 이메일 주소
const RECIPIENT_EMAIL = "@gmail.com";

// 📝 텍스트 보고서 프롬프트 (한국어/영어 모두 가능)
// 참고: 오늘 날짜는 자동으로 프롬프트에 추가되므로 "오늘"이라고 써도 됩니다.
const TEXT_PROMPT = "프롬프트들어가는자리";

// 🎨 이미지 생성 프롬프트
const IMAGE_PROMPT = "A futuristic digital art piece of a glowing currency symbol (KRW and USD) floating in a data landscape, minimalist and professional style.";

// ✉️ 이메일 제목 (날짜는 자동으로 추가됨)
const EMAIL_SUBJECT_PREFIX = "📊 [자동] 환율 전망 보고서";

// 📰 보고서 헤더 (메일 본문 상단에 표시될 제목)
const REPORT_HEADER = "🤖 제미나이 일일 환율 보고서";

// =============================================================
// 🔧 API 설정 (API CONFIG) — 일반적으로 수정 불필요
// =============================================================

const GEMINI_API_KEY = "";
const TEXT_MODEL = "gemini-3-flash-preview";
const IMAGE_MODEL = "gemini-3.1-flash-image-preview";

// =============================================================
// 이하 코드는 수정하지 않아도 됩니다
// =============================================================

/**
 * [핵심] 메인 실행 함수
 */
function sendDailyReport() {
  try {
    console.log("보고서 작성을 시작합니다...");

    // 1. 텍스트 보고서 생성 (Google Search 그라운딩 적용)
    const report = generateExchangeRateReport();
    const reportContent = report.text;
    const reportSources = report.sources;

    // 2. 이미지 생성
    const imageBlob = generateGeminiImage(IMAGE_PROMPT);

    // 3. HTML 메일 본문 구성
    let htmlBody = `
      <div style="font-family: sans-serif; line-height: 1.6; color: #333;">
        <h2 style="color: #1a73e8;">${REPORT_HEADER}</h2>
        <div style="background-color: #f8f9fa; padding: 15px; border-left: 5px solid #1a73e8;">
          <p style="white-space: pre-wrap; margin: 0;">${reportContent}</p>
        </div>
    `;

    // 출처 목록 추가 (그라운딩 결과가 있을 때만)
    if (reportSources && reportSources.length > 0) {
      htmlBody += `
        <div style="margin-top: 15px; font-size: 13px; color: #555;">
          <strong>📚 출처 (Google Search Grounding):</strong>
          <ul style="margin: 5px 0; padding-left: 20px;">
      `;
      reportSources.forEach(s => {
        htmlBody += `<li><a href="${s.uri}" style="color: #1a73e8;">${s.title}</a></li>`;
      });
      htmlBody += `</ul></div>`;
    }

    htmlBody += `<br>`;

    // 이미지 블록
    let inlineImages = {};
    if (imageBlob) {
      htmlBody += `
        <div style="text-align: center; margin-top: 20px;">
          <img src="cid:reportImage" style="width: 100%; max-width: 500px; border-radius: 8px; border: 1px solid #ddd;">
          <p style="color: #888; font-size: 12px;">Generated by Nano Banana 2</p>
        </div>
      `;
      inlineImages['reportImage'] = imageBlob;
    }

    htmlBody += `</div>`;

    // 4. Gmail 발송
    MailApp.sendEmail({
      to: RECIPIENT_EMAIL,
      subject: EMAIL_SUBJECT_PREFIX + " (" + Utilities.formatDate(new Date(), "GMT+9", "yyyy-MM-dd") + ")",
      htmlBody: htmlBody,
      inlineImages: inlineImages
    });

    console.log("보고서 발송 성공!");
  } catch (e) {
    console.error("실행 중 오류 발생: " + e.toString());
  }
}

/**
 * [함수 1] 텍스트 보고서를 생성하는 함수 (Google Search 그라운딩 적용)
 * 반환값: { text: 보고서 본문, sources: [{title, uri}, ...] }
 */
function generateExchangeRateReport() {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${TEXT_MODEL}:generateContent?key=${GEMINI_API_KEY}`;

  // 오늘 날짜를 프롬프트에 명시 (모델의 "오늘" 해석 모호성 제거)
  const today = Utilities.formatDate(new Date(), "GMT+9", "yyyy년 M월 d일");
  const groundedPrompt = `오늘은 ${today}입니다. ${today} 기준으로 답변해 주세요.\n\n${TEXT_PROMPT}`;

  const payload = {
    "contents": [{
      "parts": [{ "text": groundedPrompt }]
    }],
    // ✅ Google Search 그라운딩 활성화 (할루시네이션 방지)
    "tools": [{ "google_search": {} }],
    "generationConfig": {
      "temperature": 1.0  // 그라운딩 사용 시 권장 온도
    }
  };

  const response = UrlFetchApp.fetch(url, {
    "method": "post",
    "contentType": "application/json",
    "payload": JSON.stringify(payload),
    "muteHttpExceptions": true
  });

  const code = response.getResponseCode();
  const body = response.getContentText();

  if (code !== 200) {
    console.error("텍스트 API 오류 " + code + ": " + body);
    throw new Error("텍스트 보고서 생성 실패: " + code);
  }

  const json = JSON.parse(body);
  const text = json.candidates?.[0]?.content?.parts?.[0]?.text || "";

  // 그라운딩 출처(검색 소스) 추출
  const chunks = json.candidates?.[0]?.groundingMetadata?.groundingChunks || [];
  const sources = chunks
    .map(c => c.web)
    .filter(w => w && w.uri)
    .map(w => ({ title: w.title || w.uri, uri: w.uri }));

  return { text: text, sources: sources };
}

/**
 * [함수 2] 이미지를 생성하는 함수 (Nano Banana 2 / Gemini 3.1 Flash Image)
 */
function generateGeminiImage(prompt) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${IMAGE_MODEL}:generateContent?key=${GEMINI_API_KEY}`;

  const payload = {
    "contents": [{
      "parts": [{ "text": prompt }]
    }]
  };

  try {
    const response = UrlFetchApp.fetch(url, {
      "method": "post",
      "contentType": "application/json",
      "payload": JSON.stringify(payload),
      "muteHttpExceptions": true
    });

    const code = response.getResponseCode();
    const text = response.getContentText();

    if (code !== 200) {
      console.error("이미지 API 오류 " + code + ": " + text);
      return null;
    }

    const result = JSON.parse(text);
    const parts = result?.candidates?.[0]?.content?.parts;
    if (!parts) {
      console.warn("응답에 parts 없음: " + text);
      return null;
    }

    for (const part of parts) {
      const inline = part.inlineData || part.inline_data;
      if (inline && inline.data) {
        const bytes = Utilities.base64Decode(inline.data);
        const mime = inline.mimeType || inline.mime_type || "image/png";
        return Utilities.newBlob(bytes, mime, "generated_image.png");
      }
    }

    console.warn("inlineData 파트를 찾지 못했습니다: " + text);
  } catch (e) {
    console.warn("이미지 생성 중 예외: " + e.message);
  }
  return null;
}
